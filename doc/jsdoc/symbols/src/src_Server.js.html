1 /**
  2  * Server is an UDP server using Node's UDP API that works like the cbNetwork server.
  3  *
  4  * @requires dgram, Packet
  5  * @author Ville "tuhoojabotti" Lahdenvuo
  6  */
  7 
  8 var dgram = require('dgram'), 
  9   EventEmitter = process.EventEmitter,
 10   Packet = require('./Packet').Packet;
 11 
 12 function Client(address, port, id, sock) {
 13   this.address = address;
 14   this.port = port;
 15   this.id = id;
 16   this.data;
 17   this.sock = sock;
 18 }
 19 
 20 Client.prototype.reply = function (data) {
 21   if (!data instanceof Packet) {throw Error('Data must be a Packet object.');}
 22   data.clientId = this.id;
 23   this.sock.send(data.memBlock, 0, data.memBlock.length, this.port, this.address, function (err) {
 24     if (err) {console.log(err);}
 25   });
 26 };
 27 
 28 function Server(port, address) {
 29   var self = this;
 30   this.clients = [];
 31   this.sock = dgram.createSocket('udp4');
 32   this.sock.on('message', function (msg, peer) {
 33     var data = new Packet(new Buffer(msg)); // Get a REAL Buffer from the ugly SlowBuffer :S
 34     var client;
 35     if( data.clientId === 0 ) {
 36       client = new Client(peer.address, peer.port, self.clients.length + 1, self.sock);
 37       client.data = data;
 38       self.clients.push(client);
 39     } else {
 40       client = self.getClient(data.clientId);
 41       client.port = peer.port;
 42       client.data = data;
 43     }
 44     self.emit('message', client);
 45   });
 46   this.sock.on('error', function (e) {
 47     console.log('UDP ERROR');
 48     console.log(e);
 49     self.sock.close();
 50   });
 51   this.sock.on('close', function (e) {
 52     console.log('UDP CLOSED');
 53     console.log(e);
 54   });
 55   this.sock.bind(port, address);
 56 
 57   console.log("Server listening on " + address + ":" + port);
 58 }
 59 
 60 Server.prototype.__proto__ = EventEmitter.prototype;
 61 
 62 Server.prototype.send = function (data, address, port) {
 63   if (!data instanceof Packet) {throw Error('Data must be a Packet object.');}
 64   this.sock.send(data.memBlock, 0, data.memBlock.length, port, address, function (err) {
 65     if (err) {console.log(err);}
 66   });
 67 };
 68 
 69 Server.prototype.getClient = function (id) {
 70   for(var i = 0; i < this.clients.length; i++) {
 71     if (this.clients[i].id === id) {return this.clients[i];}
 72   }
 73 };
 74 
 75 exports.initServer = Server;
