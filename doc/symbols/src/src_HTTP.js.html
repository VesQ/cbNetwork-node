1 /**
  2  * HTTP abstraction that works like cbNetwork equivalent
  3  *
  4  * @requires http, url
  5  * @author Ville "tuhoojabotti" Lahdenvuo
  6  */
  7 
  8 /** @ignore */
  9 var http = require('http'),
 10   urlParser = require('url');
 11 
 12 /**
 13  * Creates a HTTP GET request and calls the callback with the results
 14  *
 15  * @param {String} url   Address to GET
 16  * @param {Function} cb  function (statusCode, data) { } statusCode will be 0 on failure
 17  */
 18 exports.get = function (url, cb) {
 19   var urlObj = urlParser.parse(url);
 20   http.get({
 21     host: urlObj.hostname,
 22     port: 80,
 23     path: urlObj.path,
 24     agent: false
 25   // Let's wait for the server's response
 26   }, function (res) {
 27     var data = '';
 28     // Some data received
 29     res.on('data', function (chunk) {
 30       data += chunk;
 31     });
 32     // All done
 33     res.on('end', function () {
 34       cb(res.statusCode, data);
 35     });
 36   // Something went wrong before we got our response
 37   }).on('error', function (e) {
 38     cb(0, e.message);
 39   });
 40 };
