  7 /**
  8  * Creates a new Buffer. Always define size when creating a new Packet if possible.
  9  * Writing to a buffer with undefined size causes resize function to create and copy
 10  * information to a new, bigger, buffer which is slow.
 11  *
 12  * @class Packet is used to emulate CoolBasic MemBlock-system that is used in cbNetwork for packet data.
 13  *
 14  * @param {Number|Buffer} [input]  the size in bytes to be allocated. The actual size of allocation
 15  *                                 will actually be the given size + 4 bytes because of internal use.
 16  *
 17  */
 18 function Packet (input) {
 19   if ('number' === typeof input) {
 20     this.memBlock = new Buffer(input + 4);
 21   } else if (input instanceof Buffer) {
 22     this.memBlock = input;
 23   } else if (input instanceof Packet) {
 24     this.memBlock = input.memBlock;
 25   } else {
 26     this.memBlock = new Buffer(4);
 27   }
 28   // First integer is the ID, skip it.
 29   this.offset = 4;
 30 };
 31 
 32 /**
 33  * Increases the size of the memBlock by the given one. This only allows increasing the size.
 34  *
 35  * @param {Number} size  the new size in bytes added to the offset if needed
 36  */
 37 Packet.prototype.resize = function (size) {
 38   // Only allow increasement
 39   if (this.offset + size > this.memBlock.length) {
 40     var newBuf = new Buffer(this.offset + size);
 41     this.memBlock.copy(newBuf, 0, 0);
 42     this.memBlock = newBuf;
 43   }
 44 };

 50 /**
 51  * Returns an unsigned byte from the memblock and moves the offset accordingly.
 52  *
 53  * @returns {Number}  an unsigned byte, 0...255
 54  */
 55 Packet.prototype.getByte = function () {
 56   return this.memBlock[this.offset++];
 57 };
 58 
 59 /**
 60  * Returns a signed short from the memblock and moves the offset accordingly.
 61  *
 62  * @returns {Number}  a 16-bit signed integer, -32768...32768
 63  */
 64 Packet.prototype.getShort = function () {
 65   this.offset += 2;
 66   return this.memBlock.readInt16LE(this.offset - 2);
 67 };
 68 
 69 /**
 70  * Returns an unsigned short from the memblock and moves the offset accordingly.
 71  *
 72  * @returns {Number}  a 16-bit unsigned integer, 0...65535
 73  */
 74 Packet.prototype.getUShort = function () {
 75   this.offset += 2;
 76   return this.memBlock.readUInt16LE(this.offset - 2);
 77 };
 78 
 79 /**
 80  * Returns an integer from the memblock and moves the offset accordingly.
 81  *
 82  * @returns {Number}  a 32-bit integer, -2147483647...2147483647
 83  */
 84 Packet.prototype.getInt = function () {
 85   this.offset += 4;
 86   return this.memBlock.readInt32LE(this.offset - 4);
 87 };
 88 
 89 /**
 90  * Returns a float from the memblock and moves the offset accordingly.
 91  *
 92  * @returns {Number}  a 32-bit float, 3.4e38...3.4e-38 (7 numbers)
 93  */
 94 Packet.prototype.getFloat = function () {
 95   this.offset += 4;
 96   return this.memBlock.readFloatLE(this.offset - 4);
 97 };
 98 
 99 /**
100  * Returns a string from the memblock and moves the offset accordingly.
101  *
102  * @returns {String}
103  */
104 Packet.prototype.getString = function () {
105   var len = this.getInt(), str = '';
106   for (var i = 0; i < len; i++) {
107     str += String.fromCharCode(this.memBlock[this.offset++]);
108   }
109   return str;
110 };

117 Packet.prototype.__defineGetter__('clientId', function (value) {
118   return this.memBlock.readInt32LE(0);

125 /**
126  * Puts a byte to the memblock and moves the offset accordingly.
127  *
128  * @param {Number} value  Byte to write, must be in range 0...255
129  * @throws {TypeError}    If value not a Number or not in range
130  */
131 Packet.prototype.putByte = function (value) {
132   if ('number' !== typeof value || value < 0 || value > 255) {
133     throw TypeError('Byte to write must be Number and in range 0...255');
134   }
135   this.resize(1); // Resize memBlock if needed
136   this.memBlock[this.offset++] = value;
137 };
138 
139 /**
140  * Puts a signed short to the memblock and moves the offset accordingly.
141  *
142  * @param {Number} value  a 16-bit signed integer, -32768...32768
143  * @throws {TypeError}    If value not a Number or not in range
144  */
145 Packet.prototype.putShort = function (value) {
146   if ('number' !== typeof value || value < -32768 || value > 32768) {
147     throw TypeError('Short to write must be Number and in range -32768...32768');
148   }
149   this.resize(2); // Resize memBlock if needed
150   return this.memBlock.writeInt16LE(value, this.offset);
151   this.offset += 2;
152 };
153 
154 /**
155  * Puts an unsigned short to the memblock and moves the offset accordingly.
156  *
157  * @param {Number} value  a 16-bit unsigned integer, 0...65535
158  * @throws {TypeError}    If value not a Number or not in range
159  */
160 Packet.prototype.putUShort = function (value) {
161   if ('number' !== typeof value || value < 0 || value > 65535) {
162     throw TypeError('Unsigned Short to write must be Number and in range 0...65535');
163   }
164   this.resize(2); // Resize memBlock if needed
165   return this.memBlock.writeUInt16LE(value, this.offset);
166   this.offset += 2;
167 };

169 /**
170  * Puts an integer to the memblock and moves the offset accordingly.
171  *
172  * @param {Number} value  a 32-bit integer, -2147483648...2147483648
173  * @throws {TypeError}    If value not a Number or out of range
174  */
175 Packet.prototype.putInt = function (value) {
176   if ('number' !== typeof value || value < -2147483648 || value > 2147483648) {
177     throw TypeError('Integer to write must be Number and in range -2147483648...2147483648');
178   }
179   this.resize(4); // Resize memBlock if needed
180   this.memBlock.writeInt32LE(value, this.offset);
181   this.offset += 4;
182 };

184 /**
185  * Puts a float to the memblock and moves the offset accordingly.
186  *
187  * @param {Number} value  a 32-bit float, 3.4e38...3.4e-38 (7 numbers)
188  * @throws {TypeError}    If value NaN or not a Number
189  */
190 Packet.prototype.putFloat = function (value) {
191   if ('number' !== typeof value || isNaN(value)) {
192     throw TypeError('Float to write must be Number and not NaN.');
193   }
194   this.resize(4); // Resize memBlock if needed
195   this.memBlock.writeFloatLE(value, this.offset);
196   this.offset += 4;
197 };
198 
199 /**
200  * Puts a string to the memblock and moves the offset accordingly.
201  *
202  * @param {String} value  the string to be put to the memblock
203  */
204 Packet.prototype.putString = function (value) {
205   if ('string' !== typeof value) {throw TypeError('Value must be a string');}
206   var len = value.length;
207   this.putInt(len);
208   this.resize(len); // Resize memBlock if needed
209   this.memBlock.write(value, this.offset, this.offset + len, 'ascii');
210   this.offset += len;
211 };

218  */
219 Packet.prototype.__defineSetter__('clientId', function (value) {
220   if ('number' !== typeof value || isNaN(value)) {
221     throw TypeError('ID to write must be Number and not NaN.');
222   }
223   this.memBlock.writeInt32LE(value, 0);
224 });
