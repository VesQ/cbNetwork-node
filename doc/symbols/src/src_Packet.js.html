1 /**
  2 * Packet is used to emulate CoolBasic MemBlock-system that is used in cbNetwork for packet data.
  3 */
  4 
  5 exports.Packet = Packet;
  6 
  7 /**
  8 * Creates a new Buffer.
  9 *
 10 * @param {Number|Buffer} input  the size in bytes to be allocated. The actual size of allocation
 11 *                               will actually be the given size + 4 bytes because of internal use.
 12 * @constructor
 13 */
 14 function Packet (input) {
 15   // If it's a number
 16   if ('number' === typeof input) {
 17     this.memBlock = new Buffer(input + 4);
 18   } else if (input instanceof Buffer) {
 19     this.memBlock = input;
 20   } else {
 21     this.memBlock = new Buffer(4);
 22   }
 23   this.offset = 4;
 24   this.sender = undefined;
 25 };
 26 
 27 /**
 28  * Increases the size of the memBlock by the given one. This only allows increasing the size.
 29  *
 30  * @param {Number} size  the new size in bytes added to the offset if needed
 31  */
 32 Packet.prototype.resize = function (size) {
 33   // Only allow increasement
 34   if (this.offset + size > this.memBlock.length) {
 35     var newBuf = new Buffer(this.offset + size);
 36     this.memBlock.copy(newBuf, 0, 0);
 37     this.memBlock = newBuf;
 38   }
 39 };
 40 
 41 ////////////////////////////
 42 /////////////////// GETTERS
 43 ////////////////////////////
 44 
 45 /**
 46  * Returns an unsigned byte from the memblock and moves the offset accordingly
 47  */
 48 Packet.prototype.getByte = function () {
 49   return this.memBlock[this.offset++];
 50 };
 51 
 52 /**
 53  * Returns an unsigned int from the memblock and moves the offset accordingly
 54  */
 55 Packet.prototype.getInt = function () {
 56   this.offset += 4;
 57   return this.memBlock.readUInt32LE(this.offset - 4);
 58 };
 59 
 60 /**
 61  * Returns a string from the memblock and moves the offset accordingly
 62  */
 63 Packet.prototype.getString = function () {
 64   var len = this.getInt(),
 65     str = this.memBlock.toString('ascii', this.offset, this.offset + len);
 66   this.offset += len;
 67   return str;
 68 };
 69 
 70 /**
 71  * Gets the client id, which is the first integer in memblock.
 72  */
 73 Packet.prototype.__defineGetter__('clientId', function (value) {
 74   return this.memBlock.readInt32LE(0);
 75 });

 82 /**
 83  * Puts a byte to the memblock and moves the offset accordingly
 84  *
 85  * @param {Number} value  byte to write, must be in range 0...255 
 86  */
 87 Packet.prototype.putByte = function (value) {
 88   this.resize(1); // Resize memBlock if needed
 89   this.memBlock[this.offset++] = value;
 90 };
 91 
 92 /**
 93  * Puts an int to the memblock and moves the offset accordingly
 94  *
 95  * @param {Number} value  a 32-bit integer to write to memblock
 96  */
 97 Packet.prototype.putInt = function (value) {
 98   this.resize(4); // Resize memBlock if needed
 99   this.memBlock.writeInt32LE(value, this.offset);
100   this.offset += 4;
101 };
102 
103 /**
104  * Puts a string to the memblock and moves the offset accordingly
105  *
106  * @param {String} value  the string to be put to the memblock
107  */
108 Packet.prototype.putString = function (value) {
109   if ('string' !== typeof value) {throw TypeError('Value must be a string');}
110   var len = value.length;
111   this.putInt(len);
112   this.resize(len); // Resize memBlock if needed
113   this.memBlock.write(value, this.offset, this.offset + len, 'ascii');
114   this.offset += len;
115 };
116 
117 /**
118  * Puts the client id as the first integer in the memblock.
119  */
120 Packet.prototype.__defineSetter__('clientId', function (value) {
121   this.memBlock.writeInt32LE(value, 0);
122 });
