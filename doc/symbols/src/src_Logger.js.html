1 /**
  2  * @fileOverview This file has the implementation of a Logger class for {@link cbNetwork}, but it's multipurpose.
  3  */
  4 
  5 /** @ignore */
  6 var colors = require('colors');
  7 
  8 /**
  9  * @class Creates a new logger with the desired prefix and timestamp
 10  * @requires colors
 11  *
 12  * @param {String} [prefix=[Logger] (HH:MM:SS)]   Message to apply to the start of the message %t will be replaced with a timestamp.
 13  *                                                For example: '[My logger :) [%t]]' Will print [My logger :) [00:42:30]]
 14  */
 15 function Logger(prefix) {
 16   this.prefix = prefix || '[Logger]'.cyan + '(%t)';
 17 }
 18 
 19 /**
 20  * @param {String} msg  Message to print
 21  */
 22 Logger.prototype.write = function (msg) {
 23   var s = '', d = new Date();
 24   if (this.prefix) {
 25     s = this.prefix
 26     // Apply timestamp
 27       .replace('%t',
 28         p(d.getHours())   + ':' +
 29         p(d.getMinutes()) + ':' +
 30         p(d.getSeconds()));
 31   }
 32   console.log(s + msg);
 33 };
 34 
 35 /**
 36  * @param {String} msg  Message to print green
 37  */
 38 Logger.prototype.info = function (msg) {
 39   this.write(msg.green);
 40 };
 41 
 42 /**
 43  * @param {String} msg  Message to print yellow
 44  */
 45 Logger.prototype.warn = Logger.prototype.notice = function (msg) {
 46   this.write(msg.yellow);
 47 };
 48 
 49 /**
 50  * @param {String} msg  Message to print red and bold
 51  */
 52 Logger.prototype.error = Logger.prototype.fatal = function (msg) {
 53   this.write(msg.red.bold);
 54 };
 55 
 56 /** @ignore */
 57 function p(i) {
 58   return (i < 10 ? '0' : '') + i;
 59 }
 60 
 61 exports = module.exports = Logger;
