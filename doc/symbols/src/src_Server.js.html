 12 /**
 13  * A constructor for a new client.
 14  * @private
 15  *
 16  * @class Includes client information, message and a reply function for convenience.
 17  *
 18  * @param {String} address  Client's address
 19  * @param {Number} port     Client's port
 20  * @param {Number} id       Client's unique identifier
 21  * @param {Packet} data     Client's message to the server
 22  * @param {UDP} _sock       Reference to server's UDP socket (used for answering to the client)
 23  *
 24  * @property {String} address  Client's address
 25  * @property {Number} port     Client's port
 26  * @property {String} id       Client's unique identifier, cbNetwork style
 27  * @property {Number} uid      Client's numeral id 1...2^32
 28  * @property {Packet} data     Client's message to the server
 29  * @property {UDP} _sock       Reference to server's UDP socket (used for answering to the client)
 30  */
 31 function Client(address, port, id, data, sock) {
 32   this.address = address;
 33   this.port = port;
 34   this.uid = id;
 35   this.id = address + ':' + id;
 36   this.data = data;
 37   this._sock = sock;
 38 }
 39 
 40 /**
 41  * Reply function replies to the client and automagically assigns correct client ID for the packet.
 42  *
 43  * @param {Packet} data  Data to send to client
 44  */
 45 Client.prototype.reply = function (data) {
 46   if (!data instanceof Packet) {throw Error('Data must be a Packet object.');}
 47   data.clientId = this.uid;
 48   this._sock.send(data.memBlock, 0, data.memBlock.length, this.port, this.address, function (err) {
 49     if (err) {
 50       log.error('Unable to send data to client'.red);
 51       console.dir(err);
 52     }
 53   });
 54 };

 56 /**
 57  * Creates a new server.
 58  *
 59  * @class Server is an UDP server using Node's UDP API that works like the cbNetwork server.
 60  * @requires dgram, colors, {@link Packet}
 61  *
 62  * @param {Number} port        Port to bind to.
 63  * @param {String} [address]   Address to attach to.
 64  */
 65 function Server(port, address) {
 66   var self = this;
 67   this._clients = {};
 68   this._clientCount = 0;
 69   this._sock = dgram.createSocket('udp4');
 70   this._sock.on('message', function (msg, peer) {
 71     // log.write('New packet!'.rainbow); // Awesome!
 72     var data = new Packet(new Buffer(msg)), client;
 73     if( data.clientId === 0 ) {
 74       log.info('New client connected %0 (%1)', peer.address.magenta, String(self._clientCount + 1).magenta);
 75       // Create a new client
 76       client = new Client(peer.address, peer.port, self._clientCount + 1, data, self._sock);
 77       self._clients[++self._clientCount] = client;
 78     } else {
 79       // Already exists! Update port and data
 80       client = self._clients[data.clientId];
 81       if (!client || client.address !== peer.address) {
 82         log.notice('Assigning %0 a new ID (%1)', peer.address.magenta, String(self._clientCount + 1).magenta);
 83         client = new Client(peer.address, peer.port, self._clientCount + 1, data, self._sock);
 84         self._clients[++self._clientCount] = client;
 85       }
 86       client.port = peer.port;
 87       client.data = data;
 88     }
 89     // Tell the server that we have a new message!
 90     self.emit('message', client);
 91   });
 92 
 93   // Handle errors. Neat!
 94   this._sock.on('error', function (e) {
 95     log.fatal('UDP ERROR');
 96     console.dir(e);
 97     self.close(); // Abandon ship!
 98   });
 99   this._sock.on('close', function () {
100     log.info('Server closed gracefully');
101   });
102   this._sock.on('listening', function () {
103     log.info('Server listening on %0:%1', (address ? address : '0.0.0.0').magenta, String(port).magenta);
104   });
105   // Bind the server
106   this._sock.bind(port, address);
107 }

109 Server.prototype.__proto__ = EventEmitter.prototype;

111 /** Closes the socket abandoning all clients. */
112 Server.prototype.close = function () {
113   this._sock.close();
114 };
