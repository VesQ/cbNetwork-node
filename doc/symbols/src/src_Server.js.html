1 /**
  2  * Server is an UDP server using Node's UDP API that works like the cbNetwork server.
  3  *
  4  * @requires dgram, Packet
  5  */
  6 
  7 /** @ignore */
  8 var dgram = require('dgram'),
  9   EventEmitter = process.EventEmitter,
 10   Packet = require('./Packet').Packet;
 11 
 12 /**
 13  * Client class includes client information, message and a reply function for convenience
 14  *
 15  * @constructor
 16  *
 17  * @param {String} address  Client's address
 18  * @param {Number} port     Client's port
 19  * @param {Number} id       Client's unique identifier
 20  * @param {Packet} data     Client's message to the server
 21  * @param {UDP} _sock       Reference to server's UDP socket (used for answering to the client)
 22  *
 23  * @property {String} address  Client's address
 24  * @property {Number} port     Client's port
 25  * @property {Number} id       Client's unique identifier
 26  * @property {Packet} data     Client's message to the server
 27  * @property {UDP} _sock       Reference to server's UDP socket (used for answering to the client)
 28  */
 29 function Client(address, port, id, data, sock) {
 30   this.address = address;
 31   this.port = port;
 32   this.id = id;
 33   this.data = data;
 34   this._sock = sock;
 35 }
 36 
 37 /**
 38  * Reply function replies to the client and automagically assigns correct client ID for the packet.
 39  *
 40  * @param {Packet} data  Data to send to client
 41  */
 42 Client.prototype.reply = function (data) {
 43   if (!data instanceof Packet) {throw Error('Data must be a Packet object.');}
 44   data.clientId = this.id;
 45   this._sock.send(data.memBlock, 0, data.memBlock.length, this.port, this.address, function (err) {
 46     if (err) {console.log(err);}
 47   });
 48 };
 49 
 50 /**
 51  * Server is the heart of cbNetwork-node. It creates an UDP socket and listens to it.
 52  * Automatically handling all client connections and calling 'message' event on new messages
 53  *
 54  * @constructor
 55  * @param {Number} port             Port to bind to.
 56  * @param {String} [address]        Address to attach to.
 57  */
 58 function Server(port, address) {
 59   var self = this;
 60   this._clients = {};
 61   this._clientCount = 0;
 62   this._sock = dgram.createSocket('udp4');
 63   this._sock.on('message', function (msg, peer) {
 64     var data = new Packet(new Buffer(msg)), client;
 65     if( data.clientId === 0 ) {
 66       // Create a new client
 67       client = new Client(peer.address, peer.port, self._clientCount + 1, data, self._sock);
 68       self._clients[++self._clientCount] = client;
 69     } else {
 70       // Already exists! Update port and data
 71       client = self._clients[data.clientId];
 72       client.port = peer.port;
 73       client.data = data;
 74     }
 75     // Tell the server that we have a new message!
 76     self.emit('message', client);
 77   });
 78 
 79   // Handle errors! Neat!
 80   this._sock.on('error', function (e) {
 81     console.log('UDP ERROR');
 82     console.log(e);
 83     self._sock.close();
 84   });
 85   // Handle errors! Neat!
 86   this._sock.on('close', function () {
 87     console.log('UDP CLOSED');
 88   });
 89 
 90   // Bind the server
 91   this._sock.bind(port, address);
 92   console.log("Server listening on " + address + ":" + port);
 93 }
 94 
 95 Server.prototype.__proto__ = EventEmitter.prototype;
 96 
 97 /** Closes the socket abandoning all clients. */
 98 Server.prototype.close = function () {
 99   this._sock.close();
100 };

115 
116 exports.initServer = Server;
